    84. 

  84.     function getNamespaceEnd($uri) {
    85. 

  85.         $l = strlen($uri)-1;
    86. 

  86.         do {
    87. 

  87.             $c = substr($uri, $l, 1);
    88. 

  88.             if($c == '#' || $c == ':' || $c == '/')
    89. 

  89.             break;
    90. 

  90.             $l--;
    91. 

  91.         } while ($l >= 0);
    92. 

  92.         $l++;
    93. 

  93.         return $l;
    94. 

  94.     }

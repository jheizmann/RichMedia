  171.     /**
    172. 

  172.     * Serializes a model to RDF syntax.
    173. 

  173.     * RDF syntax can be changed by config_use_attributes($boolean), config_use_entities($boolean),
    174. 

  174.     * config_sort_model($boolean).
    175. 

  175.     * NOTE: There is only one default namespace allowed within an XML document.
    176. 

  176.     *       Therefore if SER_RDF_QNAMES in constants.php is set to FALSE and you pass
    177. 

  177.     *       another $xml_default_namespace as parameter, the model will be serialized
    178. 

  178.     *       as if SER_RDF_QNAMES were set to TRUE.
    179. 

  179.     *
    180. 

  180.     * @param     object MemModel $model 
    181. 

  181.     * @param     String $encoding 
    182. 

  182.     * @return    string 
    183. 

  183.     * @access    public
    184. 

  184.     */
    185. 

  185.     function & serialize(&$model, $xml_default_namespace = NULL, $encoding = DEFAULT_ENCODING) {
    186. 

  186.  
    187. 

  187.         if ($xml_default_namespace) {
    188. 

  188.  
    189. 

  189.             if ($xml_default_namespace == RDF_NAMESPACE_URI) {
    190. 

  190.                 $this->rdf_qnames = FALSE;
    191. 

  191.                 unset($this->m_defaultNamespaces[RDF_NAMESPACE_PREFIX]);
    192. 

  192.             }
    193. 

  193.             elseif ($xml_default_namespace == RDF_SCHEMA_URI) {
    194. 

  194.                 unset($this->m_defaultNamespaces[RDF_SCHEMA_PREFIX]);
    195. 

  195.             }
    196. 

  196.             elseif (!SER_RDF_QNAMES)
    197. 

  197.             $this->rdf_qnames = TRUE;
    198. 

  198.  
    199. 

  199.             $this->addNamespacePrefix(NULL, $xml_default_namespace);
    200. 

  200.         }
    201. 

  201.  
    202. 

  202.         //copy parsed namespaces
    203. 

  203.         $nsps = array();
    204. 

  204.         $nsps = $model->getParsedNamespaces();
    205. 

  205.         foreach($this->m_defaultNamespaces as $prefix => $namespace){
    206. 

  206.             if(!isset ($nsps[$namespace]))
    207. 

  207.             $nsps[$namespace] = $prefix;
    208. 

  208.         }
    209. 

  209.  
    210. 

  210.         // define rdf prefix (qname or not)
    211. 

  211.         if ($this->rdf_qnames){
    212. 

  212.             if(isset($nsps[RDF_NAMESPACE_URI])){
    213. 

  213.                 $this->rdf_qname_prefix = $nsps[RDF_NAMESPACE_URI].':';
    214. 

  214.             }else{
    215. 

  215.                 $this->rdf_qname_prefix = RDF_NAMESPACE_PREFIX . ':';
    216. 

  216.             }
    217. 

  217.  
    218. 

  218.         }else{
    219. 

  219.             $this->rdf_qname_prefix = '';
    220. 

  220.         }
    221. 

  221.         // check if model is empty
    222. 

  222.         if ($model->size() == 0) return "<". $this->rdf_qname_prefix . RDF_RDF ." xmlns:rdf='".RDF_NAMESPACE_URI."' />";
    223. 

  223.  
    224. 

  224.         foreach($nsps as $ns => $pre){
    225. 

  225.             $this->m_namespaces[$pre] = $ns;
    226. 

  226.         }
    227. 

  227.  
    228. 

  228.  
    229. 

  229.         // set base URI
    230. 

  230.         if ($model->getBaseURI()==NULL)
    231. 

  231.         $this->m_baseURI="opaque:uri";
    232. 

  232.         else
    233. 

  233.         $this->m_baseURI=$model->getBaseURI();
    234. 

  234.  
    235. 

  235.  
    236. 

  236.         if ($this->sort_model) {
    237. 

  237.             // sort the array of statements
    238. 

  238.  
    239. 

  239.             foreach($model->triples as $key => $statement) {
    240. 

  240.                 $stmkey = $statement->subj->getURI() .
    241. 

  241.                 $statement->pred->getURI() .
    242. 

  242.                 (is_a($statement->obj,'Literal')?'"'.$statement->obj->getLabel().'"@'.$statement->obj->getLanguage().'^^'.$statement->obj->getDatatype():$statement->obj->getURI());
    243. 

  243.                 $this->m_statements[$stmkey] = $statement;
    244. 

  244.             }
    245. 

  245.             ksort($this->m_statements);
    246. 

  246.  
    247. 

  247.             /*
    248. 

  248.             // Sort using the PHP usort() function. Slower :-(
    249. 

  249.             $this->m_statements = $model->triples;
    250. 

  250.             usort($this->m_statements, "statementsorter");
    251. 

  251.             */
    252. 

  252.         } else {
    253. 

  253.             $this->m_statements = $model->triples;
    254. 

  254.         }
    255. 

  255.         // collects namespaces
    256. 

  256.         $this->m_nextAutomaticPrefixIndex=0;
    257. 

  257.         $this->collectNamespaces($model);
    258. 

  258.  
    259. 

  259.         // start writing the contents
    260. 

  260.         $this->m_out="";
    261. 

  261.         if ($this->use_xml_declaration)
    262. 

  262.         $this->m_out .= '<?xml version="1.0" encoding="' . $encoding . '" ?>' . LINEFEED;
    263. 

  263.         if (!HIDE_ADVERTISE)
    264. 

  264.         $this->m_out.="<!-- Generated by RdfSerializer.php from RDF RAP.".LINEFEED."# http://www.wiwiss.fu-berlin.de/suhl/bizer/rdfapi/index.html !-->".LINEFEED.LINEFEED ;
    265. 

  265.  
    266. 

  266.         // write entitie declarations
    267. 

  267.         if($this->use_entities) {
    268. 

  268.             $this->m_out .= '<!DOCTYPE ' . $this->rdf_qname_prefix .
    269. 

  269.             RDF_RDF.' [' . LINEFEED;
    270. 

  270.             $this->writeEntityDeclarations();
    271. 

  271.             $this->m_out .= LINEFEED . ']>' . LINEFEED;
    272. 

  272.         }
    273. 

  273.  
    274. 

  274.         // start the RDF text
    275. 

  275.         $this->m_out .= '<' . $this->rdf_qname_prefix . RDF_RDF;
    276. 

  276.  
    277. 

  277.         // write the xml:base
    278. 

  278.         if ($model->getBaseURI() != NULL)
    279. 

  279.         $this->m_out .= LINEFEED. INDENTATION .'xml:base="'.$model->getBaseURI().'"';
    280. 

  280.  
    281. 

  281.         // write namespaces declarations
    282. 

  282.         $this->writeNamespaceDeclarations();
    283. 

  283.         $this->m_out .='>'. LINEFEED;
    284. 

  284.  
    285. 

  285.         // write triples
    286. 

  286.         $this->writeDescriptions();
    287. 

  287.  
    288. 

  288.         $this->m_out .= LINEFEED;
    289. 

  289.         $this->m_out .='</' . $this->rdf_qname_prefix . RDF_RDF .'>';
    290. 

  290.  
    291. 

  291.         $this->m_namespaces=null;
    292. 

  292.         $this->m_statements=null;
    293. 

  293.         $this->m_currentSubject=null;
    294. 

  294.         $this->m_groupTypeStatement=null;
    295. 

  295.         $this->m_attributeStatements=null;
    296. 

  296.         $this->m_contentStatements=null;
    297. 

  297.         $this->m_rdfResourceElementText=null;
    298. 

  298.  
    299. 

  299.         return $this->m_out;
    300. 

  300.     }

  338.     function writeNamespaceDeclarations() {
    339. 

  339.         foreach($this->m_namespaces as $prefix => $namespace) {
    340. 

  340.  
    341. 

  341.             if ($prefix == RDF_NAMESPACE_PREFIX && !$this->rdf_qnames) {
    342. 

  342.  
    343. 

  343.                 if($this->use_entities) {
    344. 

  344.                     $this->m_out .= LINEFEED . INDENTATION .XML_NAMESPACE_DECLARATION_PREFIX .
    345. 

  345.                     '="&' . $prefix . ';"';
    346. 

  346.                 } else {
    347. 

  347.                     $this->m_out .= LINEFEED . INDENTATION .XML_NAMESPACE_DECLARATION_PREFIX .
    348. 

  348.                     '="' . $namespace . '"';
    349. 

  349.                 }
    350. 

  350.             } else {
    351. 

  351.                 if ($prefix == NULL)
    352. 

  352.                 $colon_prefix = $prefix;
    353. 

  353.                 else
    354. 

  354.                 $colon_prefix = ":" .$prefix;
    355. 

  355.  
    356. 

  356.                 if($this->use_entities) {
    357. 

  357.                     $this->m_out .= LINEFEED . INDENTATION .XML_NAMESPACE_DECLARATION_PREFIX
    358. 

  358.                     .$colon_prefix .'="&' . $prefix . ';"';
    359. 

  359.                 } else {
    360. 

  360.                     $this->m_out .= LINEFEED . INDENTATION .XML_NAMESPACE_DECLARATION_PREFIX
    361. 

  361.                     .$colon_prefix .'="' . $namespace . '"';
    362. 

  362.                 }
    363. 

  363.             }
    364. 

  364.         }
    365.

  370.     function writeDescriptions()  {
    371. 

  371.  
    372. 

  372.         $this->m_groupTypeStatement = NULL;
    373. 

  373.         $this->m_attributeStatements = array();
    374. 

  374.         $this->m_contentStatements = array();
    375. 

  375.         $this->m_currentSubject = NULL;
    376. 

  376.  
    377. 

  377.         foreach($this->m_statements as $key => $statement) {
    378. 

  378.             $subject = $statement->getSubject();
    379. 

  379.             $predicate = $statement->getPredicate();
    380. 

  380.             $object = $statement->getobject();
    381. 

  381.  
    382. 

  382.             // write Group and update current subject if nessesary
    383. 

  383.             if ($this->m_currentSubject==NULL || !$this->m_currentSubject->equals($subject)) {
    384. 

  384.                 $this->writeGroup();
    385. 

  385.                 $this->m_currentSubject=$subject;
    386. 

  386.             }
    387. 

  387.  
    388. 

  388.             // classify the statement
    389. 

  389.             if (($predicate->getURI() == RDF_NAMESPACE_URI.RDF_TYPE) && is_a($object, 'Resource')) {
    390. 

  390.                 if ($this->m_groupTypeStatement)
    391. 

  391.                 $this->writeGroup();
    392. 

  392.                 $this->m_groupTypeStatement = $statement;
    393. 

  393.             }
    394. 

  394.             elseif ($this->canAbbreviateValue($object) &&
    395. 

  395.             $this->use_attributes &&
    396. 

  396.             $this->checkForDoubleAttributes($predicate))
    397. 

  397.             {
    398. 

  398.                 if (is_a($object, 'Literal')) {
    399. 

  399.                     if ($object->getDatatype() == NULL) {
    400. 

  400.                         $this->m_attributeStatements[] = $statement;
    401. 

  401.                     } else {
    402. 

  402.                         $this->m_contentStatements[] = $statement;
    403. 

  403.                     }
    404. 

  404.                 } else {
    405. 

  405.                     $this->m_attributeStatements[] = $statement;
    406. 

  406.                 }
    407. 

  407.             } else {
    408. 

  408.                 $this->m_contentStatements[] = $statement;
    409. 

  409.             }
    410. 

  410.         }
    411. 

  411.         $this->writeGroup();
    412. 

  412.     }

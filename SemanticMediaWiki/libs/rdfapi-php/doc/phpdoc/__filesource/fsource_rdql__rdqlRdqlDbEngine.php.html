  183.  function generateSql_WhereClause($modelID) {
    184. 

  184.  
    185. 

  185.    $sql_where = ' WHERE';
    186. 

  186.    $count_patterns = count($this->parsedQuery['patterns']);
    187. 

  187.    foreach ($this->parsedQuery['patterns'] as $n => $pattern) {
    188. 

  188.      $sql_where .= ' s' .($n+1) .'.modelID=' .$modelID .' AND';
    189. 

  189.      foreach ($pattern as $key => $val_1)
    190. 

  190.        if ($val_1['value'] && $val_1['value']{0}=='?') {
    191. 

  191.          $sql_tmp = ' s' .($n+1) .'.' .$key .'=';
    192. 

  192.          // find internal bindings
    193. 

  193.          switch ($key) {
    194. 

  194.            case 'subject':
    195. 

  195.                     if ($pattern['subject']['value'] == $pattern['predicate']['value'])
    196. 

  196.                        $sql_where .= $sql_tmp .'s' .($n+1) .'.predicate AND';
    197. 

  197.                     elseif ($pattern['subject']['value'] == $pattern['object']['value'])
    198. 

  198.                        $sql_where .= $sql_tmp .'s' .($n+1) .'.object AND';
    199. 

  199.                     break;
    200. 

  200.            case 'predicate':
    201. 

  201.                     if ($pattern['predicate']['value'] == $pattern['object']['value'])
    202. 

  202.                        $sql_where .= $sql_tmp .'s' .($n+1) .'.object AND';
    203. 

  203.          }
    204. 

  204.          // find external bindings
    205. 

  205.          for ($i=$n+1; $i<$count_patterns; $i++)
    206. 

  206.              foreach ($this->parsedQuery['patterns'][$i] as $key2 => $val_2)
    207. 

  207.                if ($val_1['value']==$val_2['value']) {
    208. 

  208.                   $sql_where .= $sql_tmp .'s' .($i+1) .'.' .$key2 .' AND';
    209. 

  209.                   break 2;
    210. 

  210.                }
    211. 

  211.        }else {
    212. 

  212.           $sql_where .= ' s' .($n+1) .'.' .$key ."='" .$val_1['value'] ."' AND";
    213. 

  213.           if ($key == 'object' && isset($val_1['is_literal'])) {
    214. 

  214.               $sql_where .= ' s' .($n+1) .".object_is='l' AND";
    215. 

  215.               $sql_where .= ' s' .($n+1) .".l_datatype='" .$val_1['l_dtype'] ."' AND";
    216. 

  216.               $sql_where .= ' s' .($n+1) .".l_language='" .$val_1['l_lang'] ."' AND";     
    217. 

  217.           }
    218. 

  218.        }
    219. 

  219.    }
    220. 

  220.    return rtrim($sql_where, ' AND');
    221. 

  221.  }

  228.  *

  229.  * @param   object ADORecordSet &$recordSet 
    230. 

  230.  * @return  array  [][?VARNAME]['value']   = string
    231. 

  231.  *                              ['nType']   = string
    232. 

  232.  *                              ['l_lang']  = string
    233.

  236.  function filterQueryResult(&$recordSet) {
    237. 

  237.  
    238. 

  238.    $queryResult=array();
    239. 

  239.  
    240. 

  240.    if (isset($this->parsedQuery['filters'])) {
    241. 

  241.  
    242. 

  242.        while (!$recordSet->EOF) {
    243. 

  243.  
    244. 

  244.          foreach ($this->parsedQuery['filters'] as $filter) {
    245. 

  245.  
    246. 

  246.            $evalFilterStr = $filter['evalFilterStr'];
    247. 

  247.  
    248. 

  248.            // evaluate regex equality expressions of each filter
    249. 

  249.            foreach ($filter['regexEqExprs'] as $i => $expr) {
    250. 

  250.                preg_match($expr['regex'], $recordSet->fields[$this->rsIndexes[$expr['var']]['value']], $match);
    251. 

  251.                $op = substr($expr['operator'], 0,1);
    252. 

  252.                if (($op != '!' && !isset($match[0])) || ($op == '!' && isset($match[0])))
    253. 

  253.                   $evalFilterStr = str_replace("##RegEx_$i##", 'FALSE', $evalFilterStr);
    254. 

  254.                else
    255. 

  255.                   $evalFilterStr = str_replace("##RegEx_$i##", 'TRUE', $evalFilterStr);
    256. 

  256.            }
    257. 

  257.  
    258. 

  258.            // evaluate string equality expressions
    259. 

  259.            foreach ($filter['strEqExprs'] as $i => $expr) {
    260. 

  260.  
    261. 

  261.              $exprBoolVal = 'FALSE';
    262. 

  262.  
    263. 

  263.              switch ($expr['value_type']) {
    264. 

  264.  
    265. 

  265.                case 'variable':
    266. 

  266.                     if (($recordSet->fields[$this->rsIndexes[$expr['var']]['value']] ==
    267. 

  267.                            $recordSet->fields[$this->rsIndexes[$expr['value']]['value']] &&
    268. 

  268.                          $expr['operator'] == 'eq') ||
    269. 

  269.                         ($recordSet->fields[$this->rsIndexes[$expr['var']]['value']] !=
    270. 

  270.                            $recordSet->fields[$this->rsIndexes[$expr['value']]['value']] &&
    271. 

  271.                          $expr['operator'] == 'ne'))
    272. 

  272.                          
    273. 

  273.                        $exprBoolVal = 'TRUE';
    274. 

  274.                     break;
    275. 

  275.  
    276. 

  276.                case 'URI':
    277. 

  277.  
    278. 

  278.                       if (isset($this->rsIndexes[$expr['var']]['nType']) &&
    279. 

  279.                            $recordSet->fields[$this->rsIndexes[$expr['var']]['nType']] == 'l') {
    280. 

  280.  
    281. 

  281.                          if ($expr['operator'] == 'ne')
    282. 

  282.                             $exprBoolVal = 'TRUE';
    283. 

  283.                          break;
    284. 

  284.                       }
    285. 

  285.  
    286. 

  286.                     if (($recordSet->fields[$this->rsIndexes[$expr['var']]['value']] ==
    287. 

  287.                            $expr['value'] && $expr['operator'] == 'eq') ||
    288. 

  288.                         ($recordSet->fields[$this->rsIndexes[$expr['var']]['value']] !=
    289. 

  289.                            $expr['value'] && $expr['operator'] == 'ne'))
    290. 

  290.                        $exprBoolVal = 'TRUE';
    291. 

  291.                     break;
    292. 

  292.  
    293. 

  293.                case 'Literal':
    294. 

  294.  
    295. 

  295.                     if (!isset($this->rsIndexes[$expr['var']]['nType']) ||
    296. 

  296.                            $recordSet->fields[$this->rsIndexes[$expr['var']]['nType']] != 'l') {
    297. 

  297.  
    298. 

  298.                        if ($expr['operator'] == 'ne')
    299. 

  299.                           $exprBoolVal = 'TRUE';
    300. 

  300.                        break;
    301. 

  301.                     }
    302. 

  302.  
    303. 

  303.                     $filterLiteral= new Literal($expr['value'],$expr['value_lang']);
    304. 

  304.                     $filterLiteral->setDatatype($expr['value_dtype']);
    305. 

  305.                     
    306. 

  306.                     $resultLiteral=new Literal($recordSet->fields[$this->rsIndexes[$expr['var']]['value']]);
    307. 

  307.                     $resultLiteral->setDatatype($recordSet->fields[$this->rsIndexes[$expr['var']]['l_dtype']]);
    308. 

  308.                     $resultLiteral->setLanguage($recordSet->fields[$this->rsIndexes[$expr['var']]['l_lang']]);
    309. 

  309.                      
    310. 

  310.                     $equal=$resultLiteral->equals($filterLiteral);
    311. 

  311.                     
    312. 

  312.                     if (($equal && $expr['operator'] == 'eq') ||
    313. 

  313.                         (!$equal && $expr['operator'] == 'ne'))
    314. 

  314.                        $exprBoolVal = 'TRUE';
    315. 

  315.                     else
    316. 

  316.                        $exprBoolVal = 'FALSE';
    317. 

  317.                         
    318. 

  318.              }
    319. 

  319.  
    320. 

  320.              $evalFilterStr = str_replace("##strEqExpr_$i##", $exprBoolVal, $evalFilterStr);
    321. 

  321.           }
    322. 

  322.  
    323. 

  323.           // evaluate numerical expressions
    324. 

  324.           foreach ($filter['numExprVars'] as $varName) {
    325. 

  325.             $varValue = "'" .$recordSet->fields[$this->rsIndexes[$varName]['value']] ."'";
    326. 

  326.             $evalFilterStr = str_replace($varName, $varValue, $evalFilterStr);
    327. 

  327.           }
    328. 

  328.  
    329. 

  329.           eval("\$filterBoolVal = $evalFilterStr; \$eval_filter_ok = TRUE;");
    330. 

  330.           if (!isset($eval_filter_ok))
    331. 

  331.              trigger_error(RDQL_AND_ERR ."'" .htmlspecialchars($filter['string']) ."'", E_USER_ERROR);
    332. 

  332.  
    333. 

  333.           if (!$filterBoolVal) {
    334. 

  334.              $recordSet->MoveNext();
    335. 

  335.              continue 2;
    336. 

  336.           }
    337. 

  337.  
    338. 

  338.         }
    339. 

  339.         $queryResult[] = $this->_convertRsRowToQueryResultRow($recordSet->fields);
    340. 

  340.         $recordSet->MoveNext();
    341. 

  341.       }
    342. 

  342.         
    343. 

  343.    }else
    344. 

  344.       while (!$recordSet->EOF) {
    345. 

  345.         $queryResult[] = $this->_convertRsRowToQueryResultRow($recordSet->fields);
    346. 

  346.         $recordSet->MoveNext();
    347. 

  347.       }
    348. 

  348.    return $queryResult;
    349. 

  349.  }

  362.  function toString(&$queryResult) {
    363. 

  363.  
    364. 

  364.    // if a result set is empty return only variable sames
    365. 

  365.    if (count($queryResult) == 0) {
    366. 

  366.       foreach ($this->parsedQuery['selectVars'] as $selectVar)
    367. 

  367.          $res[0][$selectVar] = NULL;
    368. 

  368.       return $res;
    369. 

  369.    }
    370. 

  370.  
    371. 

  371.    $res = array();
    372. 

  372.    foreach ($queryResult as $n => $var)
    373. 

  373.      foreach ($var as $varname => $varProperties)
    374. 

  374.        if ($varProperties['nType'] == 'r' || $varProperties['nType'] == 'b')
    375. 

  375.           $res[$n][$varname] = '<' .$varProperties['value'] .'>';
    376. 

  376.        else {
    377. 

  377.           $res[$n][$varname] = '"' .$varProperties['value'] .'"';
    378. 

  378.           if ($varProperties['l_lang'] != NULL)
    379. 

  379.              $res[$n][$varname] .= ' (xml:lang="' .$varProperties['l_lang'] .'")';
    380. 

  380.           if ($varProperties['l_dtype'] != NULL)
    381. 

  381.              $res[$n][$varname] .= ' (rdf:datatype="' .$varProperties['l_dtype'] .'")';
    382. 

  382.        }
    383. 

  383.    return $res;
    384. 

  384.  }

    391. 

  391.  *                                            ['nType']   = string
    392. 

  392.  *                                            ['l_lang']  = string
    393. 

  393.  *                                            ['l_dtype'] = string

    397. 

  397.  function toNodes(&$queryResult) {
    398. 

  398.  
    399. 

  399.    // if a result set is empty return only variable sames
    400. 

  400.    if (count($queryResult) == 0) {
    401. 

  401.       foreach ($this->parsedQuery['selectVars'] as $selectVar)
    402. 

  402.          $res[0][$selectVar] = NULL;
    403. 

  403.       return $res;
    404. 

  404.    }
    405. 

  405.  
    406. 

  406.    $res = array();
    407. 

  407.    foreach ($queryResult as $n => $var)
    408. 

  408.      foreach ($var as $varname => $varProperties)
    409. 

  409.        if ($varProperties['nType'] == 'r')
    410. 

  410.           $res[$n][$varname] = new Resource($varProperties['value']);
    411.

  459.

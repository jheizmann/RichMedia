  97. function str2unicode_nfc($str=""){
    98. 

  98.     $result="";
    99. 

  99.     /* try to detect encoding */
    100. 

  100.     $tmp=str_replace("?", "", $str);
    101. 

  101.     if(strpos(utf8_decode($tmp), "?")===false){
    102. 

  102.         $str=utf8_decode($str);
    103. 

  103.     }
    104. 

  104.     for($i=0,$i_max=strlen($str);$i<$i_max;$i++){
    105. 

  105.         $nr=0;/* unicode dec nr */
    106. 

  106.         /* char */
    107. 

  107.         $char=$str[$i];
    108. 

  108.         /* utf8 binary */
    109. 

  109.         $utf8_char=utf8_encode($char);
    110. 

  110.         $bytes=strlen($utf8_char);
    111. 

  111.         if($bytes==1){
    112. 

  112.             /* 0####### (0-127) */
    113. 

  113.             $nr=ord($utf8_char);
    114. 

  114.         }
    115. 

  115.         elseif($bytes==2){
    116. 

  116.             /* 110##### 10###### = 192+x 128+x */
    117. 

  117.             $nr=((ord($utf8_char[0])-192)*64) + (ord($utf8_char[1])-128);
    118. 

  118.         }
    119. 

  119.         elseif($bytes==3){
    120. 

  120.             /* 1110#### 10###### 10###### = 224+x 128+x 128+x */
    121. 

  121.             $nr=((ord($utf8_char[0])-224)*4096) + ((ord($utf8_char[1])-128)*64) + (ord($utf8_char[2])-128);
    122. 

  122.         }
    123. 

  123.         elseif($bytes==4){
    124. 

  124.             /* 1111#### 10###### 10###### 10###### = 240+x 128+x 128+x 128+x */
    125. 

  125.             $nr=((ord($utf8_char[0])-240)*262144) + ((ord($utf8_char[1])-128)*4096) + ((ord($utf8_char[2])-128)*64) + (ord($utf8_char[3])-128);
    126. 

  126.         }
    127. 

  127.         /* result (see http://www.w3.org/TR/rdf-testcases/#ntrip_strings) */
    128. 

  128.         if($nr<9){/* #x0-#x8 (0-8) */
    129. 

  129.             $result.="\\u".sprintf("%04X",$nr);
    130. 

  130.         }
    131. 

  131.         elseif($nr==9){/* #x9 (9) */
    132. 

  132.             $result.='\t';
    133. 

  133.         }
    134. 

  134.         elseif($nr==10){/* #xA (10) */
    135. 

  135.             $result.='\n';
    136. 

  136.         }
    137. 

  137.         elseif($nr<13){/* #xB-#xC (11-12) */
    138. 

  138.             $result.="\\u".sprintf("%04X",$nr);
    139. 

  139.         }
    140. 

  140.         elseif($nr==13){/* #xD (13) */
    141. 

  141.             $result.='\t';
    142. 

  142.         }
    143. 

  143.         elseif($nr<32){/* #xE-#x1F (14-31) */
    144. 

  144.             $result.="\\u".sprintf("%04X",$nr);
    145. 

  145.         }
    146. 

  146.         elseif($nr<34){/* #x20-#x21 (32-33) */
    147. 

  147.             $result.=$char;
    148. 

  148.         }
    149. 

  149.         elseif($nr==34){/* #x22 (34) */
    150. 

  150.             $result.='\"';
    151. 

  151.         }
    152. 

  152.         elseif($nr<92){/* #x23-#x5B (35-91) */
    153. 

  153.             $result.=$char;
    154. 

  154.         }
    155. 

  155.         elseif($nr==92){/* #x5C (92) */
    156. 

  156.             $result.='\\';
    157. 

  157.         }
    158. 

  158.         elseif($nr<127){/* #x5D-#x7E (93-126) */
    159. 

  159.             $result.=$char;
    160. 

  160.         }
    161. 

  161.         elseif($nr<65536){/* #x7F-#xFFFF (128-65535) */
    162. 

  162.             $result.="\\u".sprintf("%04X",$nr);
    163. 

  163.         }
    164. 

  164.         elseif($nr<1114112){/* #x10000-#x10FFFF (65536-1114111) */
    165. 

  165.             $result.="\\U".sprintf("%08X",$nr);
    166. 

  166.         }
    167. 

  167.         else{
    168. 

  168.             /* other chars are not defined => ignore */
    169. 

  169.         }
    170. 

  170.     }
    171. 

  171.     return $result;
    172. 

  172. }

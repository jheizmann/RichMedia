  288.  function joinTuples(&$finalRes, &$res) {
    289. 

  289.  
    290. 

  290.    if (count($finalRes) == 0 || count($res) == 0)
    291. 

  291.       return array();
    292. 

  292.  
    293. 

  293.    // find joint variables and new variables to be added to $finalRes
    294. 

  294.    $jointVars = array();
    295. 

  295.    $newVars = array();
    296. 

  296.    foreach ($res[0] as $varname => $node) {
    297. 

  297.      if (array_key_exists($varname, $finalRes[0]))
    298. 

  298.         $jointVars[] = $varname;
    299. 

  299.      else
    300. 

  300.         $newVars[] = $varname;
    301. 

  301.    }
    302. 

  302.  
    303. 

  303.    // eliminate rows of $finalRes in which the values of $jointVars do not have
    304. 

  304.    // a corresponding row in $res.
    305. 

  305.    foreach ($finalRes as $n => $fRes) {
    306. 

  306.      foreach ($res as $i => $r) {
    307. 

  307.        $ok = TRUE;
    308. 

  308.        foreach ($jointVars as $j_varname)
    309. 

  309.          if ($r[$j_varname] != $fRes[$j_varname]) {
    310. 

  310.             $ok = FALSE;
    311. 

  311.             break;
    312. 

  312.          }
    313. 

  313.        if ($ok)
    314. 

  314.           break;
    315. 

  315.      }
    316. 

  316.      if (!$ok)
    317. 

  317.         unset($finalRes[$n]);
    318. 

  318.    }
    319. 

  319.

  320.    // join $res and $finalRes
    321. 

  321.    $joinedRes = array();
    322. 

  322.    foreach ($res as $i => $r) {
    323. 

  323.      foreach ($finalRes as $n => $fRes) {
    324. 

  324.        $ok = TRUE;
    325. 

  325.        foreach ($jointVars as $j_varname)
    326. 

  326.          if ($r[$j_varname] != $fRes[$j_varname]) {
    327. 

  327.             $ok = FALSE;
    328. 

  328.             break;
    329. 

  329.          }
    330. 

  330.        if ($ok) {
    331. 

  331.           $joinedRow = $finalRes[$n];
    332. 

  332.           foreach($newVars as $n_varname)
    333. 

  333.             $joinedRow[$n_varname] = $r[$n_varname];
    334. 

  334.           $joinedRes[] = $joinedRow;
    335. 

  335.        }
    336. 

  336.      }
    337. 

  337.    }
    338. 

  338.    
    339. 

  339.    return $joinedRes;

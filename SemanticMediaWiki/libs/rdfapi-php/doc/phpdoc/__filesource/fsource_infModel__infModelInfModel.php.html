  301.     {

  302.         $return= array();
    303. 

  303.         $statementPosition=-1;
    304. 

  304.         do
    305. 

  305.         {
    306. 

  306.             //get the position of the statement that should be removed
    307. 

  307.             $statementPosition=$this->findFirstMatchOff($statement->getSubject(),
    308. 

  308.                                                     $statement->getPredicate(),
    309. 

  309.                                                     $statement->getObject(),
    310. 

  310.                                                     $statementPosition+1);                        
    311. 

  311.             if ($statementPosition!=-1)
    312. 

  312.             {
    313. 

  313.                 //if it added any rules
    314. 

  314.                 if (isset ($this->statementRuleIndex[$statementPosition]))
    315. 

  315.                 {
    316. 

  316.                     //remove all rules 
    317. 

  317.                     foreach ($this->statementRuleIndex[$statementPosition] as $key => $value)
    318. 

  318.                     {
    319. 

  319.                         //remove from Rule-Trigger Index
    320. 

  320.                         if (is_a($this,'InfModelF'))
    321. 

  321.                         {
    322. 

  322.                             $trigger=$this->infRules[$key]->getTrigger();
    323. 

  323.                             
    324. 

  324.                             if(is_a($trigger['s'],'Node'))
    325. 

  325.                             {
    326. 

  326.                                 $subjectLabel=$trigger['s']->getLabel();
    327. 

  327.                             } else 
    328. 

  328.                             {
    329. 

  329.                                 $subjectLabel='null';
    330. 

  330.                             }
    331. 

  331.                             unset ($this->infRulesTriggerIndex['s'][$subjectLabel][array_search($key,$this->infRulesTriggerIndex['s'][$subjectLabel])]);
    332. 

  332.                             
    333. 

  333.                             if(is_a($trigger['p'],'Node'))
    334. 

  334.                             {
    335. 

  335.                                 $predicateLabel=$trigger['p']->getLabel();
    336. 

  336.                             } else 
    337. 

  337.                             {
    338. 

  338.                                 $predicateLabel='null';
    339. 

  339.                             }
    340. 

  340.                             unset ($this->infRulesTriggerIndex['p'][$predicateLabel][array_search($key,$this->infRulesTriggerIndex['p'][$predicateLabel])]);
    341. 

  341.                             
    342. 

  342.                             if(is_a($trigger['o'],'Node'))
    343. 

  343.                             {
    344. 

  344.                                 $objectLabel=$trigger['o']->getLabel();
    345. 

  345.                             } else 
    346. 

  346.                             {
    347.

  350.                         } else
    351. 

  351.                         //remove from Rule-Entailment Index
    352. 

  352.                         {    
    353.

    354. 

  354.                             
    355.

  359.                             {
    360. 

  360.                                 $subjectLabel='null';
    361. 

  361.                             }
    362. 

  362.                             unset ($this->infRulesEntailIndex['s'][$subjectLabel][array_search($key,$this->infRulesEntailIndex['s'][$subjectLabel])]);
    363. 

  363.                             
    364. 

  364.                             
    365. 

  365.                             if(is_a($entailment['p'],'Node'))
    366. 

  366.                             {
    367. 

  367.                                 $predicateLabel=$entailment['p']->getLabel();
    368. 

  368.                             } else 
    369. 

  369.                             {
    370. 

  370.                                 $predicateLabel='null';
    371. 

  371.                             }
    372. 

  372.                             unset ($this->infRulesEntailIndex['p'][$predicateLabel][array_search($key,$this->infRulesEntailIndex['p'][$predicateLabel])]);

    374. 

  374.                             if(is_a($entailment['o'],'Node'))
    375. 

  375.                             {
    376. 

  376.                                 $objectLabel=$entailment['o']->getLabel();

  377.                             } else

  380.                             }

    382.

    383. 

  383.                         //remove from statement-Rule Index

  384.                         unset ($this->infRules[$key]);
    385. 

  385.                     }
    386. 

  386.                     unset($this->statementRuleIndex[$statementPosition]);
    387. 

  387.                     $return[]=$statementPosition;
    388. 

  388.                 };
    389. 

  389.             }

  390.             
    391.

    393.

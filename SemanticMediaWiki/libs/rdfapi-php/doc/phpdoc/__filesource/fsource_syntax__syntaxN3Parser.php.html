  295. /**
    296. 

  296. * converts a string to its unicode NFC form (e.g. \uHHHH or \UHHHHHHHH).
    297. 

  297. * 
    298. 

  298. * @param String $str 
    299. 

  299. * @return String 
    300. 

  300. * @access private
    301. 

  301. *
    302. 

  302. */
    303. 

  303. function str2unicode_nfc($str=""){
    304. 

  304.     $result="";
    305. 

  305.     /* try to detect encoding */
    306. 

  306.     $tmp=str_replace("?", "", $str);
    307. 

  307.     if(strpos(utf8_decode($tmp), "?")===false){
    308. 

  308.         $str=utf8_decode($str);
    309. 

  309.     }
    310. 

  310.     for($i=0,$i_max=strlen($str);$i<$i_max;$i++){
    311. 

  311.         $nr=0;/* unicode dec nr */
    312. 

  312.         /* char */
    313. 

  313.         $char=$str[$i];
    314. 

  314.         /* utf8 binary */
    315. 

  315.         $utf8_char=utf8_encode($char);
    316. 

  316.         $bytes=strlen($utf8_char);
    317. 

  317.         if($bytes==1){
    318. 

  318.             /* 0####### (0-127) */
    319. 

  319.             $nr=ord($utf8_char);
    320. 

  320.         }
    321. 

  321.         elseif($bytes==2){
    322. 

  322.             /* 110##### 10###### = 192+x 128+x */
    323. 

  323.             $nr=((ord($utf8_char[0])-192)*64) + (ord($utf8_char[1])-128);
    324. 

  324.         }
    325. 

  325.         elseif($bytes==3){
    326. 

  326.             /* 1110#### 10###### 10###### = 224+x 128+x 128+x */
    327. 

  327.             $nr=((ord($utf8_char[0])-224)*4096) + ((ord($utf8_char[1])-128)*64) + (ord($utf8_char[2])-128);
    328. 

  328.         }
    329. 

  329.         elseif($bytes==4){
    330. 

  330.             /* 1111#### 10###### 10###### 10###### = 240+x 128+x 128+x 128+x */
    331. 

  331.             $nr=((ord($utf8_char[0])-240)*262144) + ((ord($utf8_char[1])-128)*4096) + ((ord($utf8_char[2])-128)*64) + (ord($utf8_char[3])-128);
    332. 

  332.         }
    333. 

  333.         /* result (see http://www.w3.org/TR/rdf-testcases/#ntrip_strings) */
    334. 

  334.         if($nr<9){/* #x0-#x8 (0-8) */
    335. 

  335.             $result.="\\u".sprintf("%04X",$nr);
    336. 

  336.         }
    337. 

  337.         elseif($nr==9){/* #x9 (9) */
    338. 

  338.             $result.='\t';
    339. 

  339.         }
    340. 

  340.         elseif($nr==10){/* #xA (10) */
    341. 

  341.             $result.='\n';
    342. 

  342.         }
    343. 

  343.         elseif($nr<13){/* #xB-#xC (11-12) */
    344. 

  344.             $result.="\\u".sprintf("%04X",$nr);
    345. 

  345.         }
    346. 

  346.         elseif($nr==13){/* #xD (13) */
    347. 

  347.             $result.='\t';
    348. 

  348.         }
    349. 

  349.         elseif($nr<32){/* #xE-#x1F (14-31) */
    350. 

  350.             $result.="\\u".sprintf("%04X",$nr);
    351. 

  351.         }
    352. 

  352.         elseif($nr<34){/* #x20-#x21 (32-33) */
    353. 

  353.             $result.=$char;
    354. 

  354.         }
    355. 

  355.         elseif($nr==34){/* #x22 (34) */
    356. 

  356.             $result.='\"';
    357. 

  357.         }
    358. 

  358.         elseif($nr<92){/* #x23-#x5B (35-91) */
    359. 

  359.             $result.=$char;
    360. 

  360.         }
    361. 

  361.         elseif($nr==92){/* #x5C (92) */
    362. 

  362.             $result.='\\';
    363. 

  363.         }
    364. 

  364.         elseif($nr<127){/* #x5D-#x7E (93-126) */
    365. 

  365.             $result.=$char;
    366. 

  366.         }
    367. 

  367.         elseif($nr<65536){/* #x7F-#xFFFF (128-65535) */
    368. 

  368.             $result.="\\u".sprintf("%04X",$nr);
    369. 

  369.         }
    370. 

  370.         elseif($nr<1114112){/* #x10000-#x10FFFF (65536-1114111) */
    371. 

  371.             $result.="\\U".sprintf("%08X",$nr);
    372. 

  372.         }
    373. 

  373.         else{
    374. 

  374.             /* other chars are not defined => ignore */
    375. 

  375.         }
    376. 

  376.     }
    377. 

  377.     return $result;
    378. 

  378. }

  293. /**
    294. 

  294. * Comparison function for comparing two statements.
    295. 

  295. * statementsorter() is used by the PHP function usort ( array array, callback cmp_function)
    296. 

  296. *
    297. 

  297. * @access    private
    298. 

  298. * @param        object Statement    $a 
    299. 

  299. * @param        object Statement    $b 
    300. 

  300. * @return    integer less than, equal to, or greater than zero
    301. 

  301. * @throws phpErrpr
    302. 

  302. */  
    303. 

  303. function statementsorter($a,$b) {
    304. 

  304.       //Compare subjects
    305. 

  305.       $x=$a->getSubject();
    306. 

  306.       $y=$b->getSubject();
    307. 

  307.       $r=strcmp($x->getLabel(),$y->getLabel());
    308. 

  308.       if ($r!=0) return $r;
    309. 

  309.       //Compare predicates
    310. 

  310.       $x=$a->getPredicate();
    311. 

  311.       $y=$b->getPredicate();
    312. 

  312.       $r=strcmp($x->getURI(),$y->getURI());
    313. 

  313.       if ($r!=0) return $r;
    314. 

  314.       //Final resort, compare objects
    315. 

  315.       $x=$a->getObject();
    316. 

  316.       $y=$b->getObject();
    317. 

  317.       return strcmp($x->toString(),$y->toString());
    318. 

  318. }

  12. /**
    13. 

  13. * Enhanced view of the model that is known to contain ontology data, under a
    14. 

  14. * given ontology vocabulary (such as RDFS). OntModel together with OntClass
    15. 

  15. * and OntResource provide ontology specific methods like addSubClass(),
    16. 

  16. * listSubClasses(), hasSuperProperty(), addDomain() and  listInstances(). This class does not by
    17. 

  17. * itself compute the deductive extension of the graph under the semantic
    18. 

  18. * rules of the language. Instead, we wrap an underlying model with this
    19. 

  19. * ontology interface, that presents a convenience syntax for accessing the
    20. 

  20. * language elements. Depending on the inference capability of the underlying
    21. 

  21. * model, the OntModel will appear to contain more or less triples.
    22. 

  22. * For example, if this class is used to wrap a MemModel or DBModel, only the
    23. 

  23. * relationships asserted by the document will be reported through this
    24. 

  24. * convenience API.
    25. 

  25. * Alternatively, if the OntModel wraps an InfModel (InfModelF / InfModelB),
    26. 

  26. * the inferred triples from the extension will be reported as well.

  123.     function listClasses()

  124.     {
    125. 

  125.         //get all statements, with an rdf:type as property
    126. 

  126.         $statements= $this->find(null,$this->vocabulary->TYPE(),null);
    127. 

  127.         $return = array();
    128. 

  128.         $returnIndex=array();
    129. 

  129.         foreach ($statements as $statement) 
    130. 

  130.         {
    131. 

  131.             $objectLabel=$statement->getLabelObject();
    132. 

  132.             //if it's about a typed Individual    
    133. 

  133.             if ($objectLabel!=RDF_SCHEMA_URI.RDFS_CLASS)
    134. 

  134.             {

  135.                 if (!in_array($objectLabel,$returnIndex))
    136. 

  136.                 {
    137. 

  137.                     $returnIndex[]=$objectLabel;
    138. 

  138.                     $return[]=$statement->getObject();
    139. 

  139.                 }
    140. 

  140.             } else 
    141. 

  141.             //if it's a "class1 rdf:type rdf:class" construct
    142. 

  142.             {
    143. 

  143.                 $subjectLabel=$statement->getLabelSubject();
    144. 

  144.                 if (!in_array($subjectLabel,$returnIndex))
    145. 

  145.                 {
    146. 

  146.                     $returnIndex[]=$subjectLabel;
    147. 

  147.                     $return[]=$statement->getSubject();
    148. 

  148.                 }
    149. 

  149.             }    
    150. 

  150.         }
    151. 

  151.         //find all statements about SubClassConstructs
    152. 

  152.         $statements= $this->find(null,$this->vocabulary->SUB_CLASS_OF(),null);
    153. 

  153.         foreach ($statements as $statement) 
    154. 

  154.         {
    155. 

  155.             //add the statements object to the result
    156. 

  156.             $objectLabel=$statement->getLabelObject();
    157. 

  157.             if (!in_array($objectLabel,$returnIndex))
    158. 

  158.             {
    159. 

  159.                 $returnIndex[]=$objectLabel;
    160. 

  160.                 $return[]=$statement->getObject();
    161. 

  161.             }    
    162. 

  162.         }    
    163. 

  163.         foreach ($statements as $statement)

    165.

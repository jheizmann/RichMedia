  117.  function removeComments($query) {
    118. 

  118.  
    119. 

  119.    $last = strlen($query)-1;
    120. 

  120.    $query .= ' ';
    121. 

  121.    $clean = '';
    122. 

  122.    for ($i=0; $i<=$last; $i++) {
    123. 

  123.      // don't search for comments inside a 'literal'@lang^^dtype or "literal"@lang^^dtype
    124. 

  124.      if ($query{$i} == "'" || $query{$i} == '"') {
    125. 

  125.         $quotMark = $query{$i};
    126. 

  126.         do
    127. 

  127.           $clean .= $query{$i++};
    128. 

  128.         while($i < $last && $query{$i} != $quotMark);
    129. 

  129.         $clean .= $query{$i};
    130. 

  130.         // language
    131. 

  131.         if ($query{$i+1} == '@') {
    132. 

  132.            do{
    133. 

  133.              if ($query{$i+1} == '^' && $query{$i+2} == '^')
    134. 

  134.                 break;
    135. 

  135.              $clean .= $query{++$i};
    136. 

  136.            }while ($i < $last && $query{$i} != ' '  && $query{$i} != "\t"
    137. 

  137.                               && $query{$i} != "\n" && $query{$i} != "\r");
    138. 

  138.         }
    139. 

  139.         // datatype
    140. 

  140.         if ($query{$i+1} == '^' && $query{$i+2} == '^') {
    141. 

  141.             do
    142. 

  142.               $clean .= $query{++$i};
    143. 

  143.             while ($i < $last && $query{$i} != ' '  && $query{$i} != "\t"
    144. 

  144.                              && $query{$i} != "\n" && $query{$i} != "\r" );
    145. 

  145.         }
    146. 

  146.      // don't search for comments inside an <URI> either
    147. 

  147.      }elseif ($query{$i} == '<') {
    148. 

  148.         do{
    149. 

  149.            $clean .= $query{$i++};
    150. 

  150.         }while($i < $last && $query{$i} != '>');
    151. 

  151.         $clean .= $query{$i};
    152. 

  152.      }elseif ($query{$i} == '/') {
    153. 

  153.         // clear: // comment
    154. 

  154.         if ($i < $last && $query{$i+1} == '/') {
    155. 

  155.             while($i < $last && $query{$i} != "\n" && $query{$i} != "\r")
    156. 

  156.               ++$i;
    157. 

  157.             $clean .= ' ';
    158. 

  158.         // clear: /*comment*/
    159. 

  159.         }elseif ($i < $last-2 && $query{$i+1} == '*') {
    160. 

  160.             $i += 2;
    161. 

  161.             while($i < $last  && ($query{$i} != '*' || $query{$i+1} != '/'))
    162. 

  162.               ++$i;
    163. 

  163.             if ($i >= $last && ($query{$last-1} != '*' || $query{$last} != '/'))
    164. 

  164.                trigger_error(RDQL_SYN_ERR .": unterminated comment - '*/' missing", E_USER_ERROR);
    165. 

  165.             ++$i;
    166. 

  166.         }else
    167. 

  167.           $clean .= $query{$i};
    168. 

  168.      }else
    169. 

  169.         $clean .= $query{$i};
    170. 

  170.    }
    171. 

  171.    return $clean;

  223.   function parseSelect() {
    224. 

  224.  
    225. 

  225.    $this->_clearWhiteSpaces();
    226. 

  226.  
    227. 

  227.    // Check if the queryString contains a "SELECT" token
    228.

    230. 

  230.                                   ."' - SELECT keyword expected", E_USER_ERROR);
    231. 

  231.    unset($this->tokens[key($this->tokens)]);
    232. 

  232.    $this->_clearWhiteSpaces();
    233. 

  233.  
    234. 

  234.    // Parse SELECT *
    235. 

  235.    if (current($this->tokens) == '*') {
    236. 

  236.       unset($this->tokens[key($this->tokens)]);
    237. 

  237.       $this->parsedQuery['selectVars'][0] = '*';
    238. 

  238.       $this->_clearWhiteSpaces();
    239. 

  239.       if (strcasecmp('FROM', current($this->tokens))
    240. 

  240.           && strcasecmp('SOURCE', current($this->tokens))
    241. 

  241.           && strcasecmp('WHERE', current($this->tokens)))
    242. 

  242.         trigger_error(RDQL_SYN_ERR .": '" .htmlspecialchars(current($this->tokens))
    243. 

  243.                                ."' - SOURCE or WHERE clause expected", E_USER_ERROR);
    244. 

  244.    }
    245. 

  245.  
    246. 

  246.    // Parse SELECT ?Var (, ?Var)*
    247. 

  247.    $commaExpected = FALSE;
    248. 

  248.    $comma = FALSE;
    249. 

  249.    while (current($this->tokens) != NULL) {
    250. 

  250.      $k = key($this->tokens);
    251. 

  251.      $token = $this->tokens[$k];
    252. 

  252.  
    253. 

  253.      switch ($token) {
    254. 

  254.         case ',': if (!$commaExpected)
    255. 

  255.                      trigger_error(RDQL_SEL_ERR ." ',' - unexpected comma", E_USER_ERROR);
    256. 

  256.                   $comma = TRUE;
    257. 

  257.                   $commaExpected = FALSE;
    258. 

  258.                   break;
    259. 

  259.         case '(':
    260. 

  260.         case ')': trigger_error(RDQL_SEL_ERR ." '$token' - illegal input", E_USER_ERROR);
    261. 

  261.                   break;
    262. 

  262.         default :
    263. 

  263.                   if (!strcasecmp('FROM', $token) || !strcasecmp('SOURCE', $token)) {
    264. 

  264.                      if ($comma)
    265. 

  265.                         trigger_error(RDQL_SEL_ERR ." ',' - unexpected comma", E_USER_ERROR);
    266. 

  266.                      unset($this->tokens[$k]);
    267. 

  267.                      return $this->parseFrom();
    268. 

  268.                   }elseif (!strcasecmp('WHERE', $token) && !$comma) {
    269. 

  269.                      if ($comma)
    270. 

  270.                         trigger_error(RDQL_SEL_ERR ." ',' - unexpected comma", E_USER_ERROR);
    271. 

  271.                      unset($this->tokens[$k]);
    272. 

  272.                      return $this->parseWhere();
    273. 

  273.                   }
    274. 

  274.                   if ($token{0} == '?') {
    275. 

  275.                      $this->parsedQuery['selectVars'][] = $this->_validateVar($token, RDQL_SEL_ERR);
    276. 

  276.                      $commaExpected = TRUE;
    277. 

  277.                      $comma = FALSE;
    278. 

  278.                   }else
    279. 

  279.                      trigger_error(RDQL_SEL_ERR ." '$token' - '?' missing", E_USER_ERROR);
    280. 

  280.      }
    281. 

  281.      unset($this->tokens[$k]);
    282. 

  282.      $this->_clearWhiteSpaces();
    283. 

  283.    }
    284. 

  284.    trigger_error(RDQL_SYN_ERR . ': WHERE clause missing', E_USER_ERROR);
    285. 

  285.  }

  472.  function parseFilter($n, $filter) {
    473. 

  473.  
    474. 

  474.    if ($filter == NULL)
    475. 

  475.       trigger_error(RDQL_AND_ERR ." ',' - unexpected comma", E_USER_ERROR);
    476. 

  476.    $paren = substr_count($filter, '(') - substr_count($filter, ')');
    477. 

  477.    if ($paren != 0) {
    478.

    480.

    481.

  489. /**
    490. 

  490.  * Parse expressions inside the passed filter:
    491. 

  491.  * 1)  regex equality expressions:    ?var [~~ | =~ | !~ ] REG_EX
    492. 

  492.  * 2a) string equality expressions:   ?var  [eq | ne] "literal"@lang^^dtype.
    493. 

  493.  * 2b) string equality expressions:   ?var [eq | ne] <URI> or ?var [eq | ne] prefix:local_name
    494. 

  494.  * 3)  numerical expressions: e.q.    (?var1 - ?var2)*4 >= 20
    495. 

  495.  *
    496. 

  496.  * In cases 1-2 parse each expression of the given filter into an array of variables.
    497. 

  497.  * For each parsed expression put a place holder (e.g. ##RegEx_1##) into the filterStr.
    498. 

  498.  * The RDQLengine will then replace each place holder with the outcomming boolean value
    499. 

  499.  * of the corresponding expression.
    500. 

  500.  * The remaining filterStr contains only numerical expressions and place holders.
    501. 

  501.  *
    502. 

  502.  * @param   string  $filteStr 
    503. 

  503.  * @return  array   ['string'] = string
    504. 

  504.  *                   ['evalFilterStr'] = string
    505. 

  505.  *                   ['reqexEqExprs'][]['var'] = ?VARNAME
    506. 

  506.  *                                     ['operator'] = (eq | ne)
    507. 

  507.  *                                     ['regex'] = string
    508. 

  508.  *                   ['strEqExprs'][]['var'] = ?VARNAME
    509. 

  509.  *                                  ['operator'] = (eq | ne)
    510. 

  510.  *                                  ['value'] = string
    511. 

  511.  *                                  ['value_type'] = ('variable' | 'URI' | 'QName'| 'Literal')
    512. 

  512.  *                                  ['value_lang'] = string
    513. 

  513.  *                                  ['value_dtype'] = string
    514. 

  514.  *                                    ['value_dtype_is_qname'] = boolean
    515. 

  515.  *                   ['numExpr']['vars'][] = ?VARNAME
    516. 

  516.  * @access    private
    517. 

  517.  */
    518. 

  518.  function parseExpressions($filterStr) {
    519. 

  519.  
    520. 

  520.    $parsedFilter['string'] = $filterStr;
    521. 

  521.    $parsedFilter['regexEqExprs'] = array();
    522. 

  522.    $parsedFilter['strEqExprs'] = array();
    523. 

  523.    $parsedFilter['numExprVars'] = array();
    524. 

  524.  
    525. 

  525.    // parse regex string equality expressions, e.g. ?x ~~ !//foo.com/r!i
    526. 

  526.    $reg_ex  = "/(\?[a-zA-Z0-9_]+)\s+([~!=]~)\s+(['|\"])?([^\s'\"]+)(['|\"])?/";
    527. 

  527.    preg_match_all($reg_ex, $filterStr, $eqExprs);
    528. 

  528.    foreach ($eqExprs[0] as $i => $eqExpr) {
    529. 

  529.      $this->_checkRegExQuotation($filterStr, $eqExprs[3][$i], $eqExprs[5][$i]);
    530. 

  530.      $parsedFilter['regexEqExprs'][$i]['var'] = $this->_isDefined($eqExprs[1][$i]);
    531. 

  531.      $parsedFilter['regexEqExprs'][$i]['operator'] = $eqExprs[2][$i];
    532. 

  532.      $parsedFilter['regexEqExprs'][$i]['regex'] = $eqExprs[4][$i];
    533. 

  533.  
    534. 

  534.      $filterStr = str_replace($eqExpr, " ##RegEx_$i## ", $filterStr);
    535. 

  535.    }
    536. 

  536.  
    537. 

  537.    // parse ?var  [eq | ne] "literal"@lang^^dtype
    538. 

  538.    $reg_ex  = "/(\?[a-zA-Z0-9_]+)\s+(eq|ne)\s+(\'[^\']*\'|\"[^\"]*\")";
    539. 

  539.    $reg_ex .= "(@[a-zA-Z]+)?(\^{2}\S+:?\S+)?/i";
    540. 

  540.    preg_match_all($reg_ex, $filterStr, $eqExprs);
    541. 

  541.    foreach ($eqExprs[0] as $i => $eqExpr) {
    542. 

  542.      $parsedFilter['strEqExprs'][$i]['var'] = $this->_isDefined($eqExprs[1][$i]);#
    543. 

  543.      $parsedFilter['strEqExprs'][$i]['operator'] = strtolower($eqExprs[2][$i]);
    544. 

  544.      $parsedFilter['strEqExprs'][$i]['value'] = trim($eqExprs[3][$i],"'\"");
    545. 

  545.      $parsedFilter['strEqExprs'][$i]['value_type'] = 'Literal';
    546. 

  546.      $parsedFilter['strEqExprs'][$i]['value_lang'] = substr($eqExprs[4][$i], 1);     
    547. 

  547.      $dtype = substr($eqExprs[5][$i], 2);
    548. 

  548.      if ($dtype) {
    549. 

  549.         $parsedFilter['strEqExprs'][$i]['value_dtype'] = $this->_validateUri($dtype, RDQL_AND_ERR);
    550. 

  550.         if ($dtype{0} != '<')
    551. 

  551.            $parsedFilter['strEqExprs'][$i]['value_dtype_is_qname'] = TRUE;      
    552. 

  552.      }else 
    553. 

  553.         $parsedFilter['strEqExprs'][$i]['value_dtype'] = '';
    554. 

  554.  
    555. 

  555.      $filterStr = str_replace($eqExprs[0][$i], " ##strEqExpr_$i## ", $filterStr);
    556. 

  556.    }
    557. 

  557.    
    558. 

  558.    // parse ?var [eq | ne] ?var
    559. 

  559.    $ii = count($parsedFilter['strEqExprs']);
    560. 

  560.    $reg_ex  = "/(\?[a-zA-Z0-9_]+)\s+(eq|ne)\s+(\?[a-zA-Z0-9_]+)/i";
    561. 

  561.    preg_match_all($reg_ex, $filterStr, $eqExprs);
    562. 

  562.    foreach ($eqExprs[0] as $i => $eqExpr) {
    563. 

  563.      $parsedFilter['strEqExprs'][$ii]['var'] = $this->_isDefined($eqExprs[1][$i]);
    564. 

  564.      $parsedFilter['strEqExprs'][$ii]['operator'] = strtolower($eqExprs[2][$i]);
    565. 

  565.      $parsedFilter['strEqExprs'][$ii]['value'] = $this->_isDefined($eqExprs[3][$i]);
    566. 

  566.      $parsedFilter['strEqExprs'][$ii]['value_type'] = 'variable';
    567. 

  567.  
    568. 

  568.      $filterStr = str_replace($eqExprs[0][$i], " ##strEqExpr_$ii## ", $filterStr);
    569. 

  569.      $ii++;
    570. 

  570.    }
    571. 

  571.  
    572. 

  572.    // parse ?var [eq | ne] <URI> or ?var [eq | ne] prefix:local_name
    573. 

  573.    $reg_ex  = "/(\?[a-zA-Z0-9_]+)\s+(eq|ne)\s+((<\S+>)|(\S+:\S*))/i";
    574. 

  574.    preg_match_all($reg_ex, $filterStr, $eqExprs);
    575. 

  575.    foreach ($eqExprs[0] as $i => $eqExpr) {
    576. 

  576.      $parsedFilter['strEqExprs'][$ii]['var'] = $this->_isDefined($eqExprs[1][$i]);
    577. 

  577.      $parsedFilter['strEqExprs'][$ii]['operator'] = strtolower($eqExprs[2][$i]);
    578. 

  578.      if ($eqExprs[4][$i]) {
    579. 

  579.          $parsedFilter['strEqExprs'][$ii]['value'] = trim($eqExprs[4][$i], "<>");
    580. 

  580.          $parsedFilter['strEqExprs'][$ii]['value_type'] = 'URI';
    581. 

  581.      }else if($eqExprs[5][$i]){
    582. 

  582.          $this->_validateQName($eqExprs[5][$i], RDQL_AND_ERR);
    583. 

  583.          $parsedFilter['strEqExprs'][$ii]['value'] = $eqExprs[5][$i];
    584. 

  584.          $parsedFilter['strEqExprs'][$ii]['value_type'] = 'QName';
    585. 

  585.      }
    586. 

  586.  
    587. 

  587.      $filterStr = str_replace($eqExprs[0][$i], " ##strEqExpr_$ii## ", $filterStr);
    588. 

  588.      $ii++;
    589. 

  589.    }  
    590. 

  590.    
    591. 

  591.    $parsedFilter['evalFilterStr'] = $filterStr;
    592. 

  592.  
    593. 

  593.    // all that is left are numerical expressions and place holders for the above expressions
    594. 

  594.    preg_match_all("/\?[a-zA-Z0-9_]+/", $filterStr, $vars);
    595. 

  595.    foreach ($vars[0] as $var) {
    596. 

  596.      $parsedFilter['numExprVars'][] = $this->_isDefined($var);
    597. 

  597.    }
    598. 

  598.  
    599. 

  599.    return $parsedFilter;
    600. 

  600.  }

  776.  function _validateVarUri($token) {

  777.      
    778.

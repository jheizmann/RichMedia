  8. /**
    9. 

  9. * A InfModelB extends the InfModel Class, with a backward chaining algorithm.
    10. 

  10. * Only the loaded or added base-triples are stored.
    11. 

  11. * A find-query evaluates the inference rules and recursively tries to find the statements.
    12. 

  12. * InfModelB memorises "Dead-Ends" until the next add() command, thus
    13. 

  13. * makin a second find much faster.

    15. 

  15. * WARNING: A find(null,null,null) might take very long.

    22.

    23.

  104.     /**
    105. 

  105.      * This is the main inference method of the InfModelB
    106. 

  106.      * The algorithm works as follows:
    107. 

  107.      * Find all statements in the base model, that matches the current
    108. 

  108.      * find-query.
    109. 

  109.      * Check all rules, if they are able to deliver infered statements,
    110. 

  110.      * that match the current find-query. Don't use rules with queries,
    111. 

  111.      * that lead to dead-ends and don't use a rule-query-combination that
    112. 

  112.      * was used before in this branch (ontology loops).
    113. 

  113.      * If a rule is possible do deliver such statements, get a new
    114. 

  114.      * find-query, that is possible to find those statements, that are able
    115. 

  115.      * to trigger this rule.
    116. 

  116.      * Call this _infFind method wirh the new find-query and entail the
    117. 

  117.      * resulting statements.
    118. 

  118.      * If this rule, wasn't able to return any statements with this distinct
    119. 

  119.      * query, add this combination to the dead-ends.
    120. 

  120.      * Return the statements from the base triples and those, which were infered.
    121. 

  121.      *
    122. 

  122.      * If $findOnlyFirstMatching is set to true, only the first match in
    123. 

  123.      * the base-statements is entailed an returned (used in contains() and
    124. 

  124.      * findFirstMatchingStatement() methods).
    125. 

  125.      *
    126. 

  126.      * You can set an offset to look for the first matching statement by setting the
    127. 

  127.      * $offset var.
    128. 

  128.      * 
    129. 

  129.     * It recursively searches in the statements and rules to find matching
    130. 

  130.     * statements
    131. 

  131.     *
    132. 

  132.     * @param    object Node    $subject 
    133. 

  133.     * @param    object Node    $predicate 
    134. 

  134.     * @param    object Node    $object 
    135. 

  135.     * @param    array        $searchStringIndex 
    136. 

  136.     * @param    boolean     $findOnlyFirstMatching 
    137. 

  137.     * @param    integer     $offset 
    138. 

  138.     * @param    integer     $resultCount 
    139. 

  139.     * @return    object array Statements
    140. 

  140.     * @access    private
    141. 

  141.     */
    142. 

  142.     function _infFind ($subject,$predicate,$object, $searchStringIndex, $findOnlyFirstMatching = false, $offset = 0,$resultCount = 0 )
    143. 

  143.     {
    144. 

  144.         $return=array();
    145. 

  145.         //Find all matching statements in the base statements
    146. 

  146.         $findResult=parent::find($subject,$predicate,$object);
    147. 

  147.         //For all found statements
    148. 

  148.         foreach ($findResult->triples as $statement)
    149. 

  149.             {
    150. 

  150.                 $return[]=$statement;
    151. 

  151.                 $resultCount++;
    152. 

  152.  
    153. 

  153.                 //Return, if only the firstMatchingStatement was wanted
    154. 

  154.                 if ($findOnlyFirstMatching && $resultCount > $offset)
    155. 

  155.                     return $return;
    156. 

  156.             };
    157. 

  157.             
    158. 

  158.         //Don't infer statements about the schema (rdfs:subClass, etc..)
    159. 

  159.         //is false
    160. 

  160.         if ($predicate == null || 
    161. 

  161.             (is_a($predicate,'Node') && 
    162. 

  162.             !in_array($predicate->getLabel(),$this->supportedInference))
    163. 

  163.             )
    164. 

  164.             //Check only Rules, that the EntailmentIndex returned.
    165. 

  165.                foreach ($this->_findRuleEntailmentInIndex($subject,$predicate,$object) as $ruleKey)
    166. 

  166.             {
    167. 

  167.                 $infRule=$this->infRules[$ruleKey];
    168. 

  168.                 $serializedRuleStatement=$ruleKey.serialize($subject).serialize($predicate).serialize($object);
    169. 

  169.                 //If it is to ontology loop and no dead-end
    170. 

  170.                 if (!in_array($serializedRuleStatement, $searchStringIndex) && 
    171. 

  171.                     !in_array($serializedRuleStatement, $this->findDeadEnds))
    172. 

  172.                 {    
    173. 

  173.                     //Keep this distinct rule query cobination for 
    174. 

  174.                     //this branch to detect loops
    175. 

  175.                     $searchStringIndex[]=$serializedRuleStatement;    
    176. 

  176.                     
    177. 

  177.                     //If the rule is able to deliver statements that match 
    178. 

  178.                     //this query
    179. 

  179.                     if ($infRule->checkEntailment($subject,$predicate,$object))
    180. 

  180.                     {
    181. 

  181.                         //Get a modified find-query, that matches statements, 
    182. 

  182.                         //that trigger this rule
    183. 

  183.                         $modefiedFind=$infRule->getModifiedFind($subject,$predicate,$object);
    184. 

  184.                         //Call this method with the new find-query
    185. 

  185.                         $infFindResult=$this->_infFind($modefiedFind['s'],
    186. 

  186.                                                         $modefiedFind['p'],
    187. 

  187.                                                         $modefiedFind['o'], 
    188. 

  188.                                                         $searchStringIndex, 
    189. 

  189.                                                         $findOnlyFirstMatching,
    190. 

  190.                                                         $offset,
    191. 

  191.                                                         $resultCount) ;
    192. 

  192.                         //If it deliverd statements that matches the trigger
    193. 

  193.                         if (isset($infFindResult[0]))
    194. 

  194.                         {
    195. 

  195.                             foreach ($infFindResult as $statement)
    196. 

  196.                             {    
    197. 

  197.                                 //Entail the statements and check, if they are not about the 
    198. 

  198.                                 //ontology
    199. 

  199.                                 $newStatement=$infRule->entail($statement);
    200. 

  200.                                 if (!in_array($newStatement->getLabelPredicate(),$this->supportedInference))
    201. 

  201.                                     //Check if, the entailed statements are, what we are looking for
    202. 

  202.                                     if($this->_nodeEqualsFind($subject,$newStatement->getSubject()) && 
    203. 

  203.                                         $this->_nodeEqualsFind($predicate,$newStatement->getPredicate()) && 
    204. 

  204.                                         $this->_nodeEqualsFind($object,$newStatement->getObject() ) )
    205. 

  205.                                     {
    206. 

  206.                                         //Add to results
    207. 

  207.                                         $return[]=$newStatement;
    208. 

  208.                                         $resultCount++;
    209. 

  209.                                         
    210. 

  210.                                         //or return at once
    211. 

  211.                                         if ($findOnlyFirstMatching && $resultCount > $offset)
    212. 

  212.                                             return $return;
    213. 

  213.                                     }
    214. 

  214.                                }
    215. 

  215.                         } else 
    216. 

  216.                         {
    217. 

  217.                             //If there were no results of the rule-query-combination, 
    218. 

  218.                             //mark this combination as a dead-end.
    219. 

  219.                             $this->findDeadEnds[]=$serializedRuleStatement;
    220. 

  220.                         }
    221. 

  221.                     }
    222. 

  222.                 }    
    223. 

  223.         }
    224. 

  224.         //Return the array of the found statements
    225. 

  225.         return $return;    
    226. 

  226.     }

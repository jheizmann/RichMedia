  110.  function listModels() {
    111. 

  111.  
    112. 

  112.    $recordSet =& $this->dbConn->execute("SELECT modelURI, baseURI
    113. 

  113.                                          FROM models");
    114. 

  114.    if (!$recordSet)
    115. 

  115.       echo $this->dbConn->errorMsg();
    116. 

  116.    else {
    117. 

  117.       $models = array();
    118. 

  118.       $i=0;
    119. 

  119.       while (!$recordSet->EOF) {
    120. 

  120.  
    121. 

  121.           $models[$i]['modelURI'] = $recordSet->fields[0];
    122. 

  122.           $models[$i]['baseURI'] = $recordSet->fields[1];
    123. 

  123.             
    124. 

  124.           ++$i;
    125. 

  125.           $recordSet->moveNext();
    126. 

  126.       }
    127. 

  127.       return $models;
    128. 

  128.    }

  218.  function putModel(&$model, $modelURI=NULL) {
    219. 

  219.  
    220. 

  220.    if (!$modelURI) {
    221. 

  221.       if (is_a($model, 'MemModel'))
    222. 

  222.          $modelURI = 'DbModel-' .$this->_createUniqueModelID();
    223. 

  223.       else
    224. 

  224.          $modelURI = $model->modelURI;
    225. 

  225.    }else
    226. 

  226.       if ($this->modelExists($modelURI))
    227. 

  227.          return FALSE;
    228. 

  228.  
    229. 

  229.          
    230. 

  230.    $newDbModel = $this->getNewModel($modelURI, $model->getBaseURI());

  231.    $newDbModel->addModel($model);
    232.
